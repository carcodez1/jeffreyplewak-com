import type { NextRequest } from "next/server";
import { NextResponse } from "next/server";

function base64Nonce(bytes = 16) {
  // Web Crypto is available in Next middleware runtime.
  const arr = new Uint8Array(bytes);
  crypto.getRandomValues(arr);
  // base64 without padding, URL-safe-ish is fine for nonce usage.
  return Buffer.from(arr).toString("base64").replace(/=+$/g, "");
}

function buildCsp(nonce: string) {
  // Pragmatic Next.js CSP:
  // - script-src is strict + nonce
  // - style-src allows unsafe-inline because Next injects inline <style> tags at runtime.
  //   (If you want to remove unsafe-inline from style-src, it becomes a deeper Next integration problem.)
  //
  // No third-party scripts allowed. Images allow https/data for OG, avatars, etc.
  const directives = [
    `default-src 'self'`,
    `base-uri 'self'`,
    `object-src 'none'`,
    `frame-ancestors 'none'`,
    `form-action 'self'`,
    `img-src 'self' https: data:`,
    `font-src 'self' data:`,
    `connect-src 'self' https: wss:`,
    `script-src 'self' 'nonce-${nonce}'`,
    `style-src 'self' 'unsafe-inline'`,
    `upgrade-insecure-requests`,
  ];

  return directives.join("; ");
}

export function middleware(_req: NextRequest) {
  const res = NextResponse.next();

  // Enable CSP ONLY on Vercel Production.
  // - VERCEL_ENV=production => actual production
  // - VERCEL_ENV=preview => keep dev usability for reviewers; no CSP by default
  // Optional local hardening: set NEXT_PUBLIC_ENABLE_CSP=1 and run `next start`.
  const enableCsp =
    process.env.VERCEL_ENV === "production" ||
    (process.env.NEXT_PUBLIC_ENABLE_CSP === "1" &&
      process.env.NODE_ENV === "production" &&
      !process.env.VERCEL);

  if (!enableCsp) return res;

  const nonce = base64Nonce();
  res.headers.set("x-nonce", nonce);
  res.headers.set("Content-Security-Policy", buildCsp(nonce));

  // Hardening headers (safe, non-breaking)
  res.headers.set("X-Content-Type-Options", "nosniff");
  res.headers.set("Referrer-Policy", "strict-origin-when-cross-origin");
  res.headers.set("X-Frame-Options", "DENY");

  return res;
}

// Apply to all routes.
export const config = {
  matcher: ["/((?!_next/static|_next/image|favicon.ico).*)"],
};
